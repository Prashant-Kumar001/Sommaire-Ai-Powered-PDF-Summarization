'use client';

import React, { useState } from "react";
import {
  Card,
  CardContent,
  CardDescription,
  CardHeader,
  CardTitle,
} from "../ui/card";
import { ChevronLeft, ChevronRight, Sparkles } from "lucide-react";

interface ParsedItem {
  header: string;
  content: string;
}

interface ParsedSection {
  title: string;
  content: ParsedItem[];
}

const parseSection = (section: string): ParsedSection | null => {
  const [rawTitle, ...lines] = section.split("\n").filter(Boolean);

  const title = rawTitle?.replace(/[*#]/g, "").trim();

  const content = lines
    .filter(Boolean)
    .map((line) => {
      const cleanedLine = line.replace(/[*]/g, "").trim();
      const [header, ...rest] = cleanedLine.split(":");
      return {
        header: header?.trim() || "",
        content: rest.join(":").trim(),
      };
    })
    .filter((item) => item.header && item.content); // Remove empty items

  // Return null if section has no valid content
  if (!title || content.length === 0) return null;

  return { title, content };
};

const SummaryViewer = ({
  summaryText,
  title,
}: {
  summaryText: string;
  title: string;
}) => {
  const [currentSectionIndex, setCurrentSectionIndex] = useState(0);

  if (!summaryText) return null;

  const sections = summaryText
    .split("\n\n")
    .map((s) => s.trim())
    .filter(Boolean)
    .map(parseSection)
    .filter((section): section is ParsedSection => section !== null);

  if (sections.length === 0) return null;

  const currentSection = sections[currentSectionIndex];
  const totalSections = sections.length;
  const progressPercentage = ((currentSectionIndex + 1) / totalSections) * 100;

  const handleNext = () => {
    if (currentSectionIndex < totalSections - 1) {
      setCurrentSectionIndex(currentSectionIndex + 1);
    }
  };

  const handlePrevious = () => {
    if (currentSectionIndex > 0) {
      setCurrentSectionIndex(currentSectionIndex - 1);
    }
  };

  const handleDotClick = (index: number) => {
    setCurrentSectionIndex(index);
  };

  return (
    <div className="w-full max-w-4xl mx-auto">
      <Card className="shadow-xl border-0 bg-linear-to-br from-slate-50 via-white to-cyan-50 overflow-hidden">
        <CardHeader className="bg-linear-to-r from-cyan-600 to-cyan-500 text-white pb-6">
          <div className="flex items-center gap-3">
            <div className="p-2 bg-white/20 rounded-lg backdrop-blur-sm">
              <Sparkles className="w-6 h-6" />
            </div>
            <div>
              <CardTitle className="text-3xl font-bold">{title}</CardTitle>
              <CardDescription className="text-cyan-100 text-sm mt-1">
                Organized summary insights • {totalSections} sections
              </CardDescription>
            </div>
          </div>
        </CardHeader>

        {/* Progress Bar */}
        <div className="px-6 pt-6">
          <div className="flex items-center justify-between mb-2">
            <span className="text-xs font-semibold text-slate-600 uppercase tracking-wide">
              Progress
            </span>
            <span className="text-xs font-bold text-cyan-600">
              {currentSectionIndex + 1} / {totalSections}
            </span>
          </div>
          <div className="w-full bg-slate-200 rounded-full h-2 overflow-hidden">
            <div
              className="bg-linear-to-r from-cyan-500 to-cyan-400 h-full transition-all duration-500 ease-out"
              style={{ width: `${progressPercentage}%` }}
            />
          </div>
        </div>

        {/* Section Navigation Dots */}
        <div className="px-6 pt-5 pb-6 flex justify-center gap-2">
          {sections.map((_, index) => (
            <button
              key={index}
              onClick={() => handleDotClick(index)}
              className={`transition-all duration-300 rounded-full ${
                index === currentSectionIndex
                  ? "w-8 h-2.5 bg-linear-to-r from-cyan-600 to-cyan-500 shadow-lg"
                  : "w-2.5 h-2.5 bg-slate-300 hover:bg-slate-400"
              }`}
              aria-label={`Go to section ${index + 1}`}
            />
          ))}
        </div>

        {/* Content */}
        <CardContent className="p-6">
          <div className="animate-fadeIn">
            {/* Section Title */}
            <div className="mb-6">
              <div className="flex items-center gap-3 mb-2">
                <div className="w-1 h-8 bg-linear-to-b from-cyan-500 to-cyan-400 rounded-full" />
                <h2 className="text-2xl font-bold bg-linear-to-r from-cyan-600 to-cyan-500 bg-clip-text text-transparent">
                  {currentSection.title}
                </h2>
              </div>
              <p className="text-slate-500 text-sm ml-4">
                {currentSection.content.length} key points
              </p>
            </div>

            {/* Section Content */}
            <div className="space-y-3">
              {currentSection.content.map((item, index) => (
                <div
                  key={index}
                  className="group relative bg-white p-5 rounded-xl border border-cyan-100 hover:border-cyan-300 hover:shadow-lg transition-all duration-300 hover:-translate-y-1"
                >
                  {/* Gradient overlay on hover */}
                  <div className="absolute inset-0 bg-linear-to-r from-cyan-50 to-transparent opacity-0 group-hover:opacity-100 rounded-xl transition-opacity duration-300" />

                  <div className="relative z-10">
                    <h4 className="text-cyan-700 font-semibold mb-2 flex items-center gap-2">
                      <span className="w-2 h-2 bg-cyan-500 rounded-full" />
                      {item.header}
                    </h4>
                    <p className="text-slate-700 text-sm leading-relaxed">
                      {item.content}
                    </p>
                  </div>
                </div>
              ))}
            </div>
          </div>
        </CardContent>

        {/* Navigation Footer */}
        <div className="px-6 py-6 bg-linear-to-r from-slate-50 to-cyan-50 border-t border-slate-200 flex items-center justify-between">
          <button
            onClick={handlePrevious}
            disabled={currentSectionIndex === 0}
            className="flex items-center gap-2 px-4 py-2.5 rounded-lg bg-white border border-slate-300 text-slate-700 font-medium hover:bg-slate-50 disabled:opacity-50 disabled:cursor-not-allowed transition-all duration-200 hover:shadow-md"
          >
            <ChevronLeft className="w-4 h-4" />
            <span className="hidden sm:inline">Previous</span>
          </button>

          <div className="text-sm text-slate-600 font-medium">
            Section{" "}
            <span className="text-cyan-600 font-bold">
              {currentSectionIndex + 1}
            </span>{" "}
            of <span className="text-cyan-600 font-bold">{totalSections}</span>
          </div>

          <button
            onClick={handleNext}
            disabled={currentSectionIndex === totalSections - 1}
            className="flex items-center gap-2 px-4 py-2.5 rounded-lg bg-linear-to-r from-cyan-600 to-cyan-500 text-white font-medium hover:from-cyan-700 hover:to-cyan-600 disabled:opacity-50 disabled:cursor-not-allowed transition-all duration-200 hover:shadow-lg"
          >
            <span className="hidden sm:inline">Next</span>
            <ChevronRight className="w-4 h-4" />
          </button>
        </div>
      </Card>

      {/* Add animation styles */}
      <style jsx>{`
        @keyframes fadeIn {
          from {
            opacity: 0;
            transform: translateY(10px);
          }
          to {
            opacity: 0;
            transform: translateY(0);
          }
        }
        .animate-fadeIn {
          animation: fadeIn 0.3s ease-out;
        }
      `}</style>
    </div>
  );
};

export default SummaryViewer;
